Start UI [web] provides form field components that seamlessly integrate with React Hook Form, offering type-safe validation and consistent styling.

## Overview

All form field components follow a consistent pattern:

- Full React Hook Form integration
- Automatic error display
- Accessibility features (ARIA labels, validation states)
- Consistent layout with `FormFieldContainer`
- TypeScript support for props and validation

<Note>
Form field components are located in `src/components/form/`
</Note>

## Form Field Pattern

All field components share a common architecture:

```tsx
import { useFormField } from '@/components/form/form-field';
import { useFormFieldController } from '@/components/form/form-field-controller/context';
import { FormFieldContainer } from '@/components/form/form-field-container';
import { FormFieldError } from '@/components/form/form-field-error';

export const FieldExample = (props) => {
  const ctx = useFormField();                    // Field context (id, describedBy)
  const { field, fieldState } = useFormFieldController(); // React Hook Form

  return (
    <FormFieldContainer>
      <Input
        id={ctx.id}
        aria-invalid={fieldState.invalid ? true : undefined}
        aria-describedby={ctx.describedBy(fieldState.invalid)}
        {...field}
      />
      <FormFieldError />  {/* Auto-displays validation errors */}
    </FormFieldContainer>
  );
};
```

## Text Input Fields

### FieldText

Standard text input for text, email, tel, etc.

```tsx src/components/form/field-text/index.tsx
import { Form, FormField } from '@/components/form';
import { FieldText } from '@/components/form/field-text';
import { useForm } from 'react-hook-form';
import { z } from 'zod';
import { zodResolver } from '@hookform/resolvers/zod';

const schema = z.object({
  email: z.string().email(),
  name: z.string().min(2),
});

function MyForm() {
  const form = useForm({
    resolver: zodResolver(schema),
    defaultValues: { email: '', name: '' },
  });

  return (
    <Form form={form}>
      <FormField
        control={form.control}
        name="email"
        label="Email"
      >
        <FieldText
          type="email"
          placeholder="you@example.com"
          autoComplete="email"
        />
      </FormField>

      <FormField
        control={form.control}
        name="name"
        label="Name"
        helper="Your full name"
      >
        <FieldText
          type="text"
          placeholder="John Doe"
        />
      </FormField>
    </Form>
  );
}
```

**Props**:
- `type?: string` - Input type (text, email, tel, url, etc.)
- `placeholder?: string`
- `autoComplete?: string`
- `containerProps?: object` - Props for FormFieldContainer
- All Input component props

## Form Composition

### Complete Form Example

```tsx
import { Form, FormField } from '@/components/form';
import { FieldText } from '@/components/form/field-text';
import { FieldSelect } from '@/components/form/field-select';
import { FieldCheckbox } from '@/components/form/field-checkbox';
import { Button } from '@/components/ui/button';
import { useForm } from 'react-hook-form';
import { zodResolver } from '@hookform/resolvers/zod';
import { z } from 'zod';

const formSchema = z.object({
  name: z.string().min(2, 'Name must be at least 2 characters'),
  email: z.string().email('Invalid email address'),
  role: z.string(),
  newsletter: z.boolean(),
});

type FormData = z.infer<typeof formSchema>;

export function UserForm() {
  const form = useForm<FormData>({
    resolver: zodResolver(formSchema),
    defaultValues: {
      name: '',
      email: '',
      role: '',
      newsletter: false,
    },
  });

  const onSubmit = (data: FormData) => {
    console.log(data);
  };

  return (
    <Form form={form} onSubmit={onSubmit}>
      <FormField
        control={form.control}
        name="name"
        label="Full Name"
        helper="Enter your full name"
      >
        <FieldText placeholder="John Doe" />
      </FormField>

      <FormField
        control={form.control}
        name="email"
        label="Email"
      >
        <FieldText type="email" placeholder="you@example.com" />
      </FormField>

      <FormField
        control={form.control}
        name="role"
        label="Role"
      >
        <FieldSelect
          items={[
            { label: 'Developer', value: 'dev' },
            { label: 'Designer', value: 'design' },
            { label: 'Manager', value: 'manager' },
          ]}
          placeholder="Select role"
        />
      </FormField>

      <FormField
        control={form.control}
        name="newsletter"
        label="Subscribe to newsletter"
      >
        <FieldCheckbox />
      </FormField>

      <Button type="submit" loading={form.formState.isSubmitting}>
        Submit
      </Button>
    </Form>
  );
}
```

## FormField Component

The `FormField` wrapper provides:

```tsx
import { FormField } from '@/components/form';

<FormField
  control={form.control}     // Required: form control
  name="fieldName"           // Required: field name
  label="Field Label"        // Optional: field label
  helper="Help text"         // Optional: helper text
  required                   // Optional: shows required indicator
>
  <FieldText />              // Field component
</FormField>
```

**Props**:
- `control: Control` - React Hook Form control (required)
- `name: string` - Field name (required)
- `label?: ReactNode` - Field label
- `helper?: ReactNode` - Helper text below field
- `required?: boolean` - Shows required indicator

## Custom Field Components

Create custom field components following the same pattern:

```tsx
import { useFormField } from '@/components/form/form-field';
import { useFormFieldController } from '@/components/form/form-field-controller/context';
import { FormFieldContainer } from '@/components/form/form-field-container';
import { FormFieldError } from '@/components/form/form-field-error';
import { FieldProps } from '@/components/form/types';

export const FieldCustom = (props: FieldProps<CustomProps>) => {
  const { containerProps, ...rest } = props;
  const ctx = useFormField();
  const { field, fieldState } = useFormFieldController();

  return (
    <FormFieldContainer {...containerProps}>
      <CustomInput
        id={ctx.id}
        aria-invalid={fieldState.invalid ? true : undefined}
        aria-describedby={ctx.describedBy(fieldState.invalid)}
        {...rest}
        {...field}
      />
      <FormFieldError />
    </FormFieldContainer>
  );
};
```

## Validation

### Zod Schema Validation

All field components work seamlessly with Zod:

```tsx
import { z } from 'zod';
import { zodResolver } from '@hookform/resolvers/zod';

const schema = z.object({
  email: z.string().email(),
  age: z.number().min(18).max(120),
  terms: z.boolean().refine((val) => val === true),
  tags: z.array(z.string()).min(1),
});

const form = useForm({
  resolver: zodResolver(schema),
});
```

### Custom Validation

```tsx
const schema = z.object({
  password: z.string()
    .min(8, 'Password must be at least 8 characters')
    .refine(
      (val) => /[A-Z]/.test(val),
      'Password must contain uppercase letter'
    ),
  confirmPassword: z.string(),
}).refine((data) => data.password === data.confirmPassword, {
  message: "Passwords don't match",
  path: ['confirmPassword'],
});
```
