Start UI [web] includes full internationalization (i18n) support using [i18next](https://www.i18next.com/) and [react-i18next](https://react.i18next.com/).

## Configuration

### i18next Setup

The i18n configuration is in `src/lib/i18n/config.ts`:

```typescript src/lib/i18n/config.ts
import { InitOptions } from 'i18next';
import { keys } from 'remeda';
import { DEFAULT_LANGUAGE_KEY, DEFAULT_NAMESPACE } from './constants';
import locales from '@/locales';

export const i18nConfig: InitOptions = {
  defaultNS: DEFAULT_NAMESPACE,
  ns: keys(locales[DEFAULT_LANGUAGE_KEY]),
  resources: locales,
  fallbackLng: DEFAULT_LANGUAGE_KEY,
  supportedLngs: keys(locales),
  
  detection: {
    caches: ['cookie'],
    cookieMinutes: 43200, // 30 days
    cookieOptions: { path: '/', sameSite: 'lax' },
  },

  returnNull: false,
  
  interpolation: {
    escapeValue: false, // React already escapes
  },
};
```

### Initialization

Initialize i18next with React integration:

```typescript src/lib/i18n/index.ts
import dayjs from 'dayjs';
import i18n from 'i18next';
import LanguageDetector from 'i18next-browser-languagedetector';
import { initReactI18next } from 'react-i18next';
import { i18nConfig } from './config';

i18n
  .use(LanguageDetector)
  .use(initReactI18next)
  .init(i18nConfig);

export const syncLanguage = (langKey: string) => {
  dayjs.locale(langKey);
};

i18n.on('languageChanged', (langKey) => syncLanguage(langKey));

export default i18n;
```

## Using Translations

### useTranslation Hook

The primary way to use translations in components:

```tsx
import { useTranslation } from 'react-i18next';

export function MyComponent() {
  const { t } = useTranslation(['common', 'auth']);

  return (
    <div>
      <h1>{t('common:actions.save')}</h1>
      <p>{t('auth:pageLogin.title')}</p>
    </div>
  );
}
```

### Namespace Shorthand

```tsx
const { t } = useTranslation(['auth']);

// With namespace prefix
{t('auth:pageLogin.title')}

// Without prefix (uses first namespace)
{t('pageLogin.title')}
```

### Interpolation

Pass variables to translations:

```tsx
const { t } = useTranslation();

// Translation: "Welcome back, {{name}}!"
<h1>{t('common:welcome', { name: user.name })}</h1>

// Translation: "This code expires in {{expiration}} minutes"
<p>{t('auth:pageLoginVerify.expireHint', { expiration: 5 })}</p>
```

### Trans Component

For translations with HTML or components:

```tsx
import { Trans } from 'react-i18next';

// Translation: "We sent a code to <b>{{email}}</b>"
<Trans
  t={t}
  i18nKey="auth:pageLoginVerify.description"
  values={{ email: 'user@example.com' }}
  components={{ b: <strong /> }}
/>
```

### Pluralization

```json
{
  "items": "{{count}} item",
  "items_plural": "{{count}} items"
}
```

```tsx
{t('common:items', { count: 1 })}  // "1 item"
{t('common:items', { count: 5 })}  // "5 items"
```

## Language Switching

### Language Switcher Component

Create a language switcher:

```tsx
import { useTranslation } from 'react-i18next';
import { AVAILABLE_LANGUAGES } from '@/lib/i18n/constants';
import { Select } from '@/components/ui/select';

export function LanguageSwitcher() {
  const { i18n, t } = useTranslation();

  return (
    <Select
      value={i18n.language}
      onValueChange={(lang) => i18n.changeLanguage(lang)}
      options={AVAILABLE_LANGUAGES.map(({ key, label }) => ({
        value: key,
        label: t(`common:languages.values.${key}`),
      }))}
    />
  );
}
```

### Programmatic Language Change

```tsx
import { useTranslation } from 'react-i18next';

function MyComponent() {
  const { i18n } = useTranslation();

  const switchToFrench = () => {
    i18n.changeLanguage('fr');
  };

  return (
    <button onClick={switchToFrench}>
      Switch to French
    </button>
  );
}
```

### Apply Direction

```tsx
import { useEffect } from 'react';
import { useTranslation } from 'react-i18next';
import { AVAILABLE_LANGUAGES } from '@/lib/i18n/constants';

export function LanguageProvider({ children }) {
  const { i18n } = useTranslation();

  useEffect(() => {
    const language = AVAILABLE_LANGUAGES.find(
      ({ key }) => key === i18n.language
    );

    if (language) {
      document.documentElement.lang = language.key;
      document.documentElement.dir = language.dir ?? 'ltr';
      document.documentElement.style.fontSize = 
        `${(language.fontScale ?? 1) * 100}%`;
    }
  }, [i18n.language]);

  return children;
}
```

## Type Safety

### TypeScript Configuration

Add type safety for translations:

```typescript src/lib/i18n/types.ts
import 'react-i18next';
import type locales from '@/locales';

type Locales = typeof locales;
type EnLocale = Locales['en'];

declare module 'react-i18next' {
  interface CustomTypeOptions {
    defaultNS: 'common';
    resources: EnLocale;
  }
}
```

This provides autocomplete and type checking:

```tsx
// вњ… Valid

// вќЊ TypeScript error
```

## Best Practices

<Steps>

<Step title="Use namespaces to organize translations">

```tsx
// Good: Organized by feature
const { t } = useTranslation(['auth', 'common']);

// Avoid: Single large translation file
const { t } = useTranslation();
```

</Step>

<Step title="Keep keys descriptive">

```json
// Good
{
  "pageLogin": {
    "title": "Sign in",
    "description": "Enter your email"
  }
}

// Avoid
{
  "t1": "Sign in",
  "d1": "Enter your email"
}
```

</Step>

<Step title="Use interpolation for dynamic content">

```tsx
// Good
{t('user:welcome', { name: user.name })}

// Avoid
{`Welcome ${user.name}!`}
```

</Step>

<Step title="Extract common translations">

```json common.json
{
  "actions": {
    "save": "Save",
    "cancel": "Cancel",
    "delete": "Delete"
  }
}
```

</Step>

<Step title="Always provide fallback">

```tsx
{t('key', 'Fallback text')}
```

</Step>

</Steps>

## Testing with i18n

### Mock Translations in Tests

```tsx tests/utils.tsx
import { I18nextProvider } from 'react-i18next';
import i18n from '@/lib/i18n';

export function renderWithI18n(component) {
  return render(
    <I18nextProvider i18n={i18n}>
      {component}
    </I18nextProvider>
  );
}
```

### Test Translations

```tsx
import { renderWithI18n } from 'tests/utils';

test('renders translated text', () => {
  const { getByText } = renderWithI18n(<MyComponent />);
  expect(getByText('Save')).toBeInTheDocument();
});
```
