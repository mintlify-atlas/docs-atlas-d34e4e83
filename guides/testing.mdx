Start UI [web] includes comprehensive testing setup with [Vitest](https://vitest.dev/) for unit/component tests and [Playwright](https://playwright.dev/) for end-to-end tests.

## Testing Stack

- **Vitest**: Fast unit test framework with browser mode
- **Playwright**: End-to-end testing across browsers
- **React Testing Library**: Component testing utilities (via vitest-browser-react)

## Unit Testing with Vitest

### Configuration

Vitest is configured in `vitest.config.ts`:

```typescript vitest.config.ts
import react from '@vitejs/plugin-react';
import { playwright } from '@vitest/browser-playwright';
import path from 'node:path';
import { defineConfig } from 'vitest/config';

const resolve = (filePath: string) => path.resolve(__dirname, filePath);

export default defineConfig({
  plugins: [react()],
  test: {
    projects: [
      {
        test: {
          name: 'browser',
          browser: {
            enabled: true,
            provider: playwright({
              contextOptions: {
                permissions: ['clipboard-write', 'clipboard-read'],
              },
            }),
            instances: [{ browser: 'chromium' }],
          },
          include: ['src/**/*.browser.{test,spec}.?(c|m)[jt]s?(x)'],
          setupFiles: [
            resolve('src/tests/setup.base.ts'),
            resolve('src/tests/setup.browser.ts'),
          ],
        },
      },
      {
        test: {
          name: 'unit',
          environment: 'node',
          include: ['src/**/*.unit.{test,spec}.?(c|m)[jt]s?(x)'],
          setupFiles: [
            resolve('src/tests/setup.base.ts'),
            resolve('src/server/routers/test-setup.ts'),
          ],
        },
      },
    ],
  },
});
```

## Browser Tests

### Component Testing

Test React components in a real browser:

```tsx src/components/button/button.browser.test.tsx
import { expect, test } from 'vitest/browser';
import { render, screen } from '@/tests/utils';
import { Button } from './button';

test('renders button with text', async () => {
  render(<Button>Click me</Button>);
  await expect.element(screen.getByText('Click me')).toBeVisible();
});

test('handles click events', async () => {
  const handleClick = vi.fn();
  render(<Button onClick={handleClick}>Click me</Button>);
  
  await screen.getByText('Click me').click();
  expect(handleClick).toHaveBeenCalledOnce();
});

test('shows loading state', async () => {
  render(<Button loading>Submit</Button>);
  await expect.element(screen.getByRole('status')).toBeVisible();
});
```

### Test Utilities

Custom render function with providers:

```tsx src/tests/utils.tsx
import { ReactElement } from 'react';
import { userEvent } from 'vitest/browser';
import { ComponentRenderOptions, render } from 'vitest-browser-react';
import { Providers } from '@/providers';

const WithProviders = ({ children }: { children: React.ReactNode }) => {
  return <Providers>{children}</Providers>;
};

const customRender = (
  ui: ReactElement,
  options?: Omit<ComponentRenderOptions, 'wrapper'>
) => {
  return render(ui, { wrapper: WithProviders, ...options });
};

export * from 'vitest/browser';
export * from 'vitest-browser-react';
export { customRender as render };
export const setupUser = () => userEvent.setup();
```

### User Interactions

```tsx
import { render, screen, setupUser } from '@/tests/utils';

test('form submission', async () => {
  const user = setupUser();
  const handleSubmit = vi.fn();
  
  render(<LoginForm onSubmit={handleSubmit} />);
  
  // Type into input
  await user.type(
    screen.getByLabelText('Email'),
    'user@example.com'
  );
  
  // Click button
  await user.click(screen.getByRole('button', { name: 'Submit' }));
  
  expect(handleSubmit).toHaveBeenCalledWith({
    email: 'user@example.com',
  });
});
```

## Unit Tests

### Server-side Testing

Test API routes and server functions:

```typescript src/server/routers/user.unit.spec.ts
import { expect, test, describe } from 'vitest';
import { orpcTestCaller } from './test-utils';
import { db } from '@/server/db';

describe('User Router', () => {
  test('creates a user', async () => {
    const caller = orpcTestCaller({ isAdmin: true });
    
    const user = await caller.user.create({
      name: 'John Doe',
      email: 'john@example.com',
      role: 'user',
    });
    
    expect(user.name).toBe('John Doe');
    expect(user.email).toBe('john@example.com');
  });

  test('lists all users', async () => {
    const caller = orpcTestCaller({ isAdmin: true });
    
    const users = await caller.user.getAll();
    
    expect(Array.isArray(users)).toBe(true);
    expect(users.length).toBeGreaterThan(0);
  });

  test('throws error for duplicate email', async () => {
    const caller = orpcTestCaller({ isAdmin: true });
    
    await caller.user.create({
      name: 'Jane Doe',
      email: 'jane@example.com',
      role: 'user',
    });

    await expect(
      caller.user.create({
        name: 'Jane Smith',
        email: 'jane@example.com',
        role: 'user',
      })
    ).rejects.toThrow();
  });
});
```

## End-to-End Testing with Playwright

### Configuration

Playwright is configured in `playwright.config.ts`:

```typescript playwright.config.ts
import { defineConfig, devices } from '@playwright/test';
import { DEFAULT_LANGUAGE_KEY } from '@/lib/i18n/constants';

export default defineConfig({
  testDir: './e2e',
  globalTimeout: 15 * 60 * 1000,
  fullyParallel: true,
  forbidOnly: !!process.env.CI,
  retries: process.env.CI ? 2 : 0,
  workers: process.env.CI ? 1 : undefined,
  reporter: process.env.CI ? 'github' : 'html',
  
  use: {
    baseURL: process.env.VITE_BASE_URL,
    trace: 'on-first-retry',
  },

  projects: [
    {
      name: 'setup',
      testMatch: /.*\.setup\.ts/,
      use: { locale: DEFAULT_LANGUAGE_KEY },
    },
    {
      name: 'chromium',
      use: { ...devices['Desktop Chrome'], locale: DEFAULT_LANGUAGE_KEY },
      dependencies: process.env.CI ? ['setup'] : [],
    },
    {
      name: 'firefox',
      use: { ...devices['Desktop Firefox'], locale: DEFAULT_LANGUAGE_KEY },
      dependencies: process.env.CI ? ['setup'] : [],
    },
    {
      name: 'webkit',
      use: { ...devices['Desktop Safari'], locale: DEFAULT_LANGUAGE_KEY },
      dependencies: process.env.CI ? ['setup'] : [],
    },
  ],

  webServer: {
    command: 'pnpm dev',
    url: process.env.VITE_BASE_URL,
    reuseExistingServer: !process.env.CI,
  },
});
```

### Login Flow Test

Real example from `e2e/login.spec.ts`:

```typescript e2e/login.spec.ts
import { expect, test } from 'e2e/utils';
import { ADMIN_EMAIL, USER_EMAIL } from 'e2e/utils/constants';

test.describe('Login flow', () => {
  test('Login as admin', async ({ page }) => {
    await page.to('/login');
    await page.login({ email: ADMIN_EMAIL });
    await page.waitForURL('/manager');
    await expect(page.getByTestId('layout-manager')).toBeVisible();
  });

  test('Login as user', async ({ page }) => {
    await page.to('/login');
    await page.login({ email: USER_EMAIL });
    await page.waitForURL('/app');
    await expect(page.getByTestId('layout-app')).toBeVisible();
  });

  test('Login with redirect', async ({ page }) => {
    await page.to('/app');
    await page.login({ email: ADMIN_EMAIL });
    await page.waitForURL('/app');
    await expect(page.getByTestId('layout-app')).toBeVisible();
  });
});
```

### Custom Test Utilities

Extend Playwright with custom helpers:

```typescript e2e/utils/page.ts
import { Page } from '@playwright/test';
import { AUTH_EMAIL_OTP_MOCKED } from '@/features/auth/config';

export async function login(
  page: Page,
  { email }: { email: string }
) {
  // Fill email
  await page.getByPlaceholder('Email').fill(email);
  await page.getByRole('button', { name: /continue/i }).click();
  
  // Enter OTP
  await page.waitForURL(/\/login\/verify/);
  const otpInput = page.getByLabel('Verification Code');
  await otpInput.fill(AUTH_EMAIL_OTP_MOCKED);
}

export async function createUser(
  page: Page,
  { name, email, role }: { name: string; email: string; role: string }
) {
  await page.to('/manager/users/new');
  await page.getByLabel('Name').fill(name);
  await page.getByLabel('Email').fill(email);
  await page.getByLabel('Role').selectOption(role);
  await page.getByRole('button', { name: 'Create' }).click();
}
```

### Testing User Flows

```typescript e2e/users.spec.ts
import { expect, test } from 'e2e/utils';
import { ADMIN_EMAIL } from 'e2e/utils/constants';

test.describe('User Management', () => {
  test.beforeEach(async ({ page }) => {
    await page.to('/login');
    await page.login({ email: ADMIN_EMAIL });
  });

  test('creates a new user', async ({ page }) => {
    await page.createUser({
      name: 'Test User',
      email: 'test@example.com',
      role: 'user',
    });

    await expect(page.getByText('User created successfully')).toBeVisible();
  });

  test('edits user details', async ({ page }) => {
    await page.to('/manager/users');
    await page.getByRole('row', { name: /Test User/ })
      .getByRole('button', { name: 'Edit' })
      .click();
    
    await page.getByLabel('Name').fill('Updated Name');
    await page.getByRole('button', { name: 'Save' }).click();

    await expect(page.getByText('User updated successfully')).toBeVisible();
  });

  test('deletes a user', async ({ page }) => {
    await page.to('/manager/users');
    await page.getByRole('row', { name: /Test User/ })
      .getByRole('button', { name: 'Delete' })
      .click();
    
    await page.getByRole('button', { name: 'Confirm' }).click();

    await expect(page.getByText('User deleted successfully')).toBeVisible();
  });
});
```

## Best Practices

<Steps>

<Step title="Use data-testid for stable selectors">

```tsx
<div data-testid="layout-manager">
  {/* content */}
</div>
```

```typescript
await expect(page.getByTestId('layout-manager')).toBeVisible();
```

</Step>

<Step title="Test user behavior, not implementation">

```tsx
// Good: Test what user sees
await expect(screen.getByText('Welcome')).toBeVisible();

// Avoid: Test internal state
expect(component.state.isVisible).toBe(true);
```

</Step>

<Step title="Use realistic test data">

```typescript
const testUser = {
  name: 'John Doe',
  email: 'john@example.com',
  role: 'user',
};
```

</Step>

<Step title="Clean up after tests">

```typescript
test.afterEach(async () => {
  await db.user.deleteMany({
    where: { email: { contains: 'test' } },
  });
});
```

</Step>

<Step title="Use fixtures for common setup">

```typescript
const test = baseTest.extend({
  authenticatedPage: async ({ page }, use) => {
    await page.login({ email: ADMIN_EMAIL });
    await use(page);
  },
});
```

</Step>

</Steps>

## CI/CD Integration

### GitHub Actions

```yaml .github/workflows/test.yml
name: Tests

on: [push, pull_request]

jobs:
  test:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v3
      - uses: actions/setup-node@v3
        with:
          node-version: 22
      
      - name: Install dependencies
        run: pnpm install
      
      - name: Run unit tests
        run: pnpm test:ci
      
      - name: Run E2E tests
        run: pnpm e2e
        env:
          VITE_BASE_URL: http://localhost:3000
```
