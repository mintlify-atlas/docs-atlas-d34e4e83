Start UI [web] includes [Storybook](https://storybook.js.org/) for component development, documentation, and visual testing.

## Running Storybook

<CodeGroup>

```bash Development
pnpm storybook
```

```bash Build
pnpm storybook:build
```

</CodeGroup>

Storybook runs at `http://localhost:6006`

## Configuration

### Main Configuration

Storybook is configured in `.storybook/main.ts`:

```typescript .storybook/main.ts
import type { StorybookConfig } from '@storybook/react-vite';

const config: StorybookConfig = {
  stories: ['../**/*.stories.@(js|jsx|mjs|ts|tsx)'],
  addons: [
    '@vueless/storybook-dark-mode',
    '@storybook/addon-a11y',
    '@storybook/addon-docs',
  ],
  framework: {
    name: '@storybook/react-vite',
    options: {
      builder: {
        viteConfigPath: './vite.storybook.ts',
      },
    },
  },
};

export default config;
```

### Preview Configuration

Global decorators and parameters in `.storybook/preview.tsx`:

```tsx .storybook/preview.tsx
import type { Preview } from '@storybook/react-vite';
import { useDarkMode } from '@vueless/storybook-dark-mode';
import { StrictMode, useEffect } from 'react';
import { useTranslation } from 'react-i18next';

import '@/styles/app.css';
import './preview.css';

import {
  AVAILABLE_LANGUAGES,
  DEFAULT_LANGUAGE_KEY,
} from '@/lib/i18n/constants';
import i18nGlobal from '@/lib/i18n';
import { Providers } from '@/providers';

const preview: Preview = {
  parameters: {
    layout: 'padded',
    backgrounds: { disable: true },
    darkMode: { stylePreview: true },
    docs: { codePanel: true },
  },
  
  initialGlobals: {
    locale: DEFAULT_LANGUAGE_KEY,
  },
  
  globalTypes: {
    locale: {
      name: 'Locale',
      description: 'Internationalization locale',
      toolbar: {
        icon: 'globe',
        items: AVAILABLE_LANGUAGES.map(({ key }) => ({
          value: key,
          title: i18nGlobal.t(`common:languages.values.${key}`, {
            lng: 'en',
          }),
        })),
      },
    },
  },
  
  decorators: [
    (story, context) => {
      const isDarkMode = useDarkMode();
      const { i18n } = useTranslation();

      useEffect(() => {
        i18n.changeLanguage(context.globals.locale);
        const languageConfig = AVAILABLE_LANGUAGES.find(
          ({ key }) => key === context.globals.locale
        );
        if (languageConfig) {
          document.documentElement.lang = languageConfig.key;
          document.documentElement.dir = languageConfig.dir ?? 'ltr';
          document.documentElement.style.fontSize = 
            `${(languageConfig.fontScale ?? 1) * 100}%`;
        }
      }, [context.globals.locale, i18n]);

      return (
        <Providers forcedTheme={isDarkMode ? 'dark' : 'light'}>
          <StrictMode>
            {story(context)}
          </StrictMode>
        </Providers>
      );
    },
  ],
};

export default preview;
```

## Writing Stories

### Basic Story

Create a story file for a component:

```tsx src/components/ui/button.stories.tsx
import type { Meta, StoryObj } from '@storybook/react-vite';
import { Button } from './button';

export default {
  title: 'Components/Button',
  component: Button,
} satisfies Meta<typeof Button>;

type Story = StoryObj<typeof Button>;

export const Default: Story = {
  args: {
    children: 'Click me',
  },
};

export const Primary: Story = {
  args: {
    variant: 'primary',
    children: 'Primary Button',
  },
};

export const Secondary: Story = {
  args: {
    variant: 'secondary',
    children: 'Secondary Button',
  },
};

export const Loading: Story = {
  args: {
    loading: true,
    children: 'Loading...',
  },
};

export const Disabled: Story = {
  args: {
    disabled: true,
    children: 'Disabled',
  },
};
```

### Form Stories

Real example from `src/components/form/docs.stories.tsx`:

```tsx src/components/form/docs.stories.tsx
import { zodResolver } from '@hookform/resolvers/zod';
import { Meta } from '@storybook/react-vite';
import { useForm } from 'react-hook-form';
import { z } from 'zod';
import { zu } from '@/lib/zod/zod-utils';
import {
  Form,
  FormField,
  FormFieldController,
  FormFieldLabel,
  FormFieldHelper,
} from '@/components/form';
import { Button } from '@/components/ui/button';

export default {
  title: 'Form/Form',
} satisfies Meta<typeof Form>;

const zFormSchema = () =>
  z.object({
    name: zu.fieldText.required(),
    other: zu.fieldText.required(),
  });

export const Default = () => {
  const form = useForm({
    mode: 'onBlur',
    resolver: zodResolver(zFormSchema()),
    defaultValues: {
      name: '',
      other: '',
    },
  });

  return (
    <Form {...form} onSubmit={console.log}>
      <div className="flex flex-col gap-4">
        <FormField size="lg">
          <FormFieldLabel>Name</FormFieldLabel>
          <FormFieldController 
            control={form.control} 
            type="text" 
            name="name" 
          />
          <FormFieldHelper>This is a helper text</FormFieldHelper>
        </FormField>
        
        <div>
          <Button type="submit">Submit</Button>
        </div>
      </div>
    </Form>
  );
};
```

## Features

### Dark Mode

Toggle dark mode with the toolbar button. Components automatically adapt:

```tsx
export const DarkModeExample = () => {
  return (
    <div className="p-4 bg-background text-foreground">
      <h1 className="text-2xl font-bold">Dark Mode Support</h1>
      <p className="text-muted-foreground">This text adapts to theme</p>
    </div>
  );
};
```

### Internationalization

Switch languages using the locale toolbar:

```tsx
import { useTranslation } from 'react-i18next';

export const TranslatedComponent = () => {
  const { t } = useTranslation();
  
  return (
    <div>
      <h1>{t('common:welcome')}</h1>
      <p>{t('common:description')}</p>
    </div>
  );
};
```

### Accessibility Testing

The a11y addon automatically checks accessibility:

```tsx
export const AccessibleButton = () => {
  return (
    <button aria-label="Close dialog">
      ×
    </button>
  );
};
```

View violations in the Accessibility panel.

## Documentation

### MDX Stories

Write documentation with MDX:

```mdx src/components/ui/button.stories.mdx
import { Meta, Canvas, Story } from '@storybook/blocks';
import { Button } from './button';

<Meta title="Components/Button" component={Button} />

# Button

Buttons allow users to take actions with a single tap.

<Canvas>
  <Story name="Basic">
    <Button>Click me</Button>
  </Story>
</Canvas>

## Variants

<Canvas>
  <Story name="Variants">
    <div className="flex gap-2">
      <Button variant="primary">Primary</Button>
      <Button variant="secondary">Secondary</Button>
    </div>
  </Story>

```

### JSDoc Comments

Document component props:

```tsx
interface ButtonProps {
  /**
   * Button variant style
   * @default 'primary'
   */
  variant?: 'primary' | 'secondary' | 'outline';
  
  /**
   * Show loading spinner
   */
  loading?: boolean;
  
  /**
   * Disable the button
   */
  disabled?: boolean;
}
```

Props appear in the Storybook Docs tab.

## Advanced Patterns

### Args with Actions

```tsx
import { action } from '@storybook/addon-actions';

export const WithAction: Story = {
  args: {
    onClick: action('clicked'),
    children: 'Click to see action',
  },
};
```

### Controls

```tsx
export default {
  title: 'Components/Button',
  component: Button,
  argTypes: {
    variant: {
      control: 'select',
      options: ['primary', 'secondary', 'outline'],
    },
    size: {
      control: 'radio',
      options: ['sm', 'md', 'lg'],
    },
    disabled: {
      control: 'boolean',
    },
  },
} satisfies Meta<typeof Button>;
```

### Play Function

Test interactions:

```tsx
import { userEvent, within } from '@storybook/test';

export const WithInteraction: Story = {
  play: async ({ canvasElement }) => {
    const canvas = within(canvasElement);
    const button = canvas.getByRole('button');
    await userEvent.click(button);
  },
};
```

## Building and Deployment

### Build Storybook

```bash
pnpm storybook:build
```

Outputs to `public/storybook/`

### Deploy to Production

The built Storybook is accessible at `/storybook` in your deployed app.

### Static Deployment

Deploy Storybook separately:

```bash
pnpm storybook:build
# Deploy storybook-static/ to hosting
```

## Troubleshooting

### Stories Not Showing

Ensure the file matches the glob pattern:

```typescript
stories: ['../**/*.stories.@(js|jsx|mjs|ts|tsx)']
```

### Styling Issues

Import styles in `.storybook/preview.tsx`:

```tsx
import '@/styles/app.css';
```

### Provider Issues

Wrap stories with necessary providers:

```tsx
decorators: [
  (story) => (
    <Providers>
      {story()}
    </Providers>
  ),
],
```
