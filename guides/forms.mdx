Start UI [web] provides a comprehensive form system built on [React Hook Form](https://react-hook-form.com/) with [Zod](https://zod.dev/) schema validation.

## Form Components

### Basic Form Setup

The `Form` component wraps React Hook Form with automatic submission handling:

```tsx
import { useForm } from 'react-hook-form';
import { zodResolver } from '@hookform/resolvers/zod';
import { Form, FormField, FormFieldController, FormFieldLabel } from '@/components/form';
import { Button } from '@/components/ui/button';
import { z } from 'zod';

const schema = z.object({
  name: z.string().min(1, 'Name is required'),
  email: z.string().email('Invalid email'),
});

export function MyForm() {
  const form = useForm({
    resolver: zodResolver(schema),
    defaultValues: {
      name: '',
      email: '',
    },
  });

  const onSubmit = async (data) => {
    console.log(data);
  };

  return (
    <Form {...form} onSubmit={onSubmit}>
      <FormField>
        <FormFieldLabel>Name</FormFieldLabel>
        <FormFieldController
          control={form.control}
          type="text"
          name="name"
        />
      </FormField>

      <Button type="submit">
        Submit
      </Button>
    </Form>
  );
}
```

### Form Component Props

The `Form` component accepts:

- All React Hook Form props from `useForm()`
- `onSubmit`: Handler function for form submission
- `noHtmlForm`: Opt-out of automatic form element wrapping

```tsx
// Automatic form element (default)
<Form {...form} onSubmit={handleSubmit}>
  {/* Fields */}
</Form>

// Manual form element
<Form {...form} noHtmlForm>
  <form onSubmit={form.handleSubmit(handleSubmit)}>
    {/* Fields */}
  </form>
</Form>
```

### Field Validation Patterns

<CodeGroup>

```typescript Email
const schema = z.object({
  email: z
    .string()
    .min(1, 'Email is required')
    .email('Invalid email address'),
});
```

```typescript Password
const schema = z.object({
  password: z
    .string()
    .min(8, 'Password must be at least 8 characters')
    .regex(
      /^(?=.*[a-z])(?=.*[A-Z])(?=.*\d)/,
      'Password must contain uppercase, lowercase, and number'
    ),
});
```

```typescript Confirm Password
const schema = z
  .object({
    password: z.string().min(8),
    confirmPassword: z.string(),
  })
  .refine((data) => data.password === data.confirmPassword, {
    message: "Passwords don't match",
    path: ['confirmPassword'],
  });
```

```typescript Custom Validation
const schema = z.object({
  username: z
    .string()
    .min(3)
    .refine(
      async (username) => {
        // Check if username is available
        const available = await checkUsername(username);
        return available;
      },
      { message: 'Username is already taken' }
    ),
});
```

## Form Patterns from the Starter

### User Creation Form

Real example from `src/features/user/manager/page-user-new.tsx`:

```tsx
import { zodResolver } from '@hookform/resolvers/zod';
import { useForm } from 'react-hook-form';
import { useMutation, useQueryClient } from '@tanstack/react-query';
import { toast } from 'sonner';

export const PageUserNew = () => {
  const { t } = useTranslation(['user']);
  const queryClient = useQueryClient();
  
  const form = useForm({
    resolver: zodResolver(zFormFieldsUser()),
    values: {
      name: '',
      email: '',
      role: 'user',
    },
  });

  const userCreate = useMutation(
    orpc.user.create.mutationOptions({
      onSuccess: async () => {
        // Invalidate users list
        await queryClient.invalidateQueries({
          queryKey: orpc.user.getAll.key(),
        });
        navigateBack({ ignoreBlocker: true });
      },
      onError: (error) => {
        if (error.code === 'CONFLICT' && error.data?.target?.includes('email')) {
          form.setError('email', {
            message: t('user:manager.form.emailAlreadyExist'),
          });
          return;
        }
        toast.error(t('user:manager.new.createError'));
      },
    })
  );

  return (
    <Form {...form} onSubmit={(values) => userCreate.mutate(values)}>
      <FormUser />
      <Button type="submit" loading={userCreate.isPending}>
        Create User
      </Button>
    </Form>
  );
};
```

### Login Form with OTP

Real example from `src/features/auth/page-login.tsx`:

```tsx
import { zodResolver } from '@hookform/resolvers/zod';
import { useForm } from 'react-hook-form';
import { authClient } from '@/features/auth/client';

export default function PageLogin({ search }) {
  const form = useForm({
    mode: 'onSubmit',
    resolver: zodResolver(zFormFieldsLogin()),
    defaultValues: { email: '' },
  });

  const submitHandler = async ({ email }) => {
    const result = await authClient.emailOtp.sendVerificationOtp({
      email,
      type: 'sign-in',
    });

    if (result.error) {
      toast.error(errorMessage);
      return;
    }

    router.navigate({
      to: '/login/verify',
      search: { redirect: search.redirect, email },

    <Form {...form} onSubmit={submitHandler}>

        <FormFieldLabel className="sr-only">

        loading={form.formState.isSubmitting}
        type="submit"
        size="lg"
      >
        Continue
      </Button>
    </Form>
  );
}
```

## Advanced Patterns

### Custom Field Component

Create custom fields with full control:

```tsx
<FormFieldController
  control={form.control}
  name="customField"
  type="custom"
  render={({ field, fieldState }) => (
    <>
      <InputGroup>
        <InputGroupAddon>
          <InputGroupText>https://</InputGroupText>
        </InputGroupAddon>
        <InputGroupInput
          {...field}
          aria-invalid={fieldState.invalid ? true : undefined}
          value={field.value ?? ''}
        />
      </InputGroup>
      <FormFieldError />
    </>
  )}
/>
```

### Form Mode Configuration

```tsx
const form = useForm({
  mode: 'onBlur',      // Validate on blur
  mode: 'onChange',    // Validate on every change
  mode: 'onSubmit',    // Validate only on submit (default)
  mode: 'onTouched',   // Validate after first blur, then on change
});
```

### Conditional Fields

```tsx
function MyForm() {
  const form = useForm();
  const showAdvanced = form.watch('showAdvanced');

  return (
    <Form {...form} onSubmit={handleSubmit}>
      <FormFieldController
        control={form.control}
        type="checkbox"
        name="showAdvanced"
        label="Show advanced options"
      />

      {showAdvanced && (
        <FormFieldController
          control={form.control}
          type="text"
          name="advancedOption"
        />
      )}
    </Form>
  );
}
```

### Dynamic Field Arrays

```tsx
import { useFieldArray } from 'react-hook-form';

function MyForm() {
  const form = useForm();
  const { fields, append, remove } = useFieldArray({
    control: form.control,
    name: 'items',
  });

  return (
    <Form {...form} onSubmit={handleSubmit}>
      {fields.map((field, index) => (
        <div key={field.id}>
          <FormFieldController
            control={form.control}
            type="text"
            name={`items.${index}.name`}
          />
          <Button onClick={() => remove(index)}>Remove</Button>
        </div>
      ))}
      <Button onClick={() => append({ name: '' })}>
        Add Item
      </Button>
    </Form>
  );
}
```

### Server Errors

Set errors from server responses:

```tsx
const onSubmit = async (data) => {

    await createUser(data);
  } catch (error) {
    if (error.code === 'CONFLICT' && error.target === 'email') {
      form.setError('email', {
        message: 'Email already exists',
      });
    }
  }
};
```

### Prevent Navigation

Warn users about unsaved changes:

```tsx
import { PreventNavigation } from '@/components/prevent-navigation';

function MyForm() {
  const form = useForm();

  return (
    <>
      <PreventNavigation shouldBlock={form.formState.isDirty} />
      <Form {...form} onSubmit={handleSubmit}>
        {/* Fields */}
      </Form>
    </>
  );
}
```

## Form State

### Accessing Form State

```tsx

const {
  isSubmitting,   // Form is currently submitting
  isValid,        // All fields are valid
  isDirty,        // Any field has been modified
  isSubmitted,    // Form has been submitted
  errors,         // Validation errors
} = form.formState;
```
